from decimal import Decimal, ROUND_DOWN

def calculate_position_size(account_usdt: float, entry_zone: list, stop_loss: float, 
                            risk_pct: float, leverage_cap: int, fee_rate: float=0.0004) -> dict:
    """
    确定性仓位计算器 - LLM仅声明意图，本函数执行精确计算
    """
    risk_amount = Decimal(str(account_usdt)) * Decimal(str(risk_pct))
    price_diff = abs(Decimal(str(entry_zone[0])) - Decimal(str(stop_loss)))
    
    if price_diff == 0:
        raise ValueError("Stop loss equals entry - invalid trade setup")
    
    # 基础名义价值 = 风险金额 / 价格差比例
    notional = risk_amount / (price_diff / Decimal(str(entry_zone[0])))
    
    # 杠杆适配：取建议上限与计算值的较小者
    leverage = min(leverage_cap, int(notional / Decimal(str(account_usdt))))
    leverage = max(1, leverage)  # 至少1x
    
    # 实际保证金 = 名义价值/杠杆 - 双边手续费预估
    actual_position = (notional / leverage) * (1 - fee_rate * 2)
    
    return {
        "notional_usdt": float(notional.quantize(Decimal('0.01'), rounding=ROUND_DOWN)),
        "leverage": leverage,
        "margin_usdt": float(actual_position.quantize(Decimal('0.01'), rounding=ROUND_DOWN)),
        "risk_amount_usdt": float(risk_amount),
        "position_size_pct": round(float(risk_pct * 100), 2)
    }

📍 风险具象化表现
场景	问题	损失机制
插针行情	市价单滑点>1%，直接击穿止损线	止损变市价，亏损放大
流动性枯竭	大单无法成交，部分成交导致风控模型失效	仓位暴露，风险失控
OCO失效	止盈止损未正确挂载，价格快速波动时无法保护利润	盈利回吐或亏损扩大
🧠 根本原因
交易所API与LLM之间存在"完美执行假设"
实盘市场存在订单簿深度、买卖价差、延迟、手续费、部分成交等物理约束
LLM无法感知微观市场结构，需代码层补足
🛡️ 架构解法：流动性感知执行路由（Liquidity-Aware Router）

# executor.py 订单路由核心逻辑
def route_order(self, decision: dict) -> dict:
    """
    流动性感知订单路由 - 根据市场状态动态选择执行策略
    """
    depth_usdt = self.get_orderbook_depth(decision['entry_zone'])
    spread = self.get_spread()
    atr = self.get_atr('15m')
    
    # 流动性不足或价差过大时，启用保守执行模式
    if depth_usdt < decision['notional_usdt'] * 3 or spread > 0.001:
        order_type = "LIMIT_POST_ONLY"
        execution_mode = "TWAP"  # 拆单执行，降低市场冲击
        time_horizon = "15m"     # TWAP执行窗口
    else:
        order_type = "LIMIT"
        execution_mode = "DIRECT"
        time_horizon = "immediate"
        
    return {
        "order_type": order_type,
        "slippage_tolerance": 0.003,  # 0.3%最大容忍滑点
        "execution_mode": execution_mode,
        "time_horizon": time_horizon,
        "reduce_only": False,
        "oco_params": {
            "stop_loss": decision['stop_loss'],
            "take_profit_1": decision['targets'][0]['price'],
            "take_profit_2": decision['targets'][1]['price'],
            "stop_trigger": "last_price"  # 使用最新价触发，避免标记价格偏差
        }
    }

# src/utils/cot_validator.py
def validate_cot(cot_text: str, decision: dict, market_context: dict) -> dict:
    """
    CoT质量校验器 - 确保推理过程逻辑完整、数值有效、结论明确
    """
    checks = {
        # 1. 步骤完整性：是否包含关键推理环节
        "has_invalidation": "失效条件" in cot_text or "invalidation" in cot_text.lower(),
        "has_risk_reward": "风险收益" in cot_text or "risk_reward" in cot_text.lower(),
        "has_entry_logic": "入场" in cot_text and ("支撑" in cot_text or "压力" in cot_text),
        
        # 2. 数值有效性：关键价格是否合理
        "stop_beyond_entry": (
            decision['stop_loss'] < min(decision['entry_zone']) 
            if decision['action']=='long' 
            else decision['stop_loss'] > max(decision['entry_zone'])
        ),
        "rr_ratio_valid": decision.get('risk_reward_ratio', 0) >= 2.0,
        
        # 3. 逻辑一致性：避免模糊结论与后见之明
        "no_hedging": not any(word in cot_text for word in ["也许", "可能涨也可能跌", "不好说", "事后看"]),
        "no_hindsight": "果然" not in cot_text and "正如预期" not in cot_text,  # 避免结果导向
        
        # 4. 置信度校准：低置信必须对应保守决策
        "confidence_action_aligned": (
            decision['confidence'] < 0.4 and decision['action'] == 'no_trade'
        ) or decision['confidence'] >= 0.4
    }
    
    # 计算质量评分
    score = sum(checks.values()) / len(checks)
    
    return {
        "passed": score >= 0.75,
        "details": checks,
        "quality_score": round(score, 2),
        "failed_reasons": [k for k, v in checks.items() if not v]
    }

🎯 开发期核心纪律（务必刻入代码规范）
1.LLM不碰确定性计算
所有涉及资金、杠杆、仓位、手续费的数值必须由Python工具层计算并二次校验。LLM仅声明"意图"，不输出"绝对值"。
2."不做交易"是最高级决策
当事实锚点冲突、CoT质量不达标、辩论无实质分歧时，系统必须优先返回 no_trade。宁缺毋滥。
3.硬风控不可覆盖
Phase 3 的6条风控规则必须写在 risk_manager.py 最底层，使用 assert 或异常机制实现。LLM输出仅为"建议"，代码拥有最终否决权。
4.日志即资产
chain_of_thought 不得截断，必须带 call_id、phase、validation_score。这是后续蒸馏与提示词优化的唯一真相源。建议每日备份至对象存储。
5.变更可追溯
所有Prompt修改必须通过YAML版本管理，每次更新附带 change_reason 与 expected_impact。禁止直接修改线上提示词。
